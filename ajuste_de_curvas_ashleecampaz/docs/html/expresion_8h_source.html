27/*
28Shunting Yard Algorithm by Edsger Wybe Dijkstra (E. W. Dijkstra)
29(Wikipedia Version)
30https://en.wikipedia.org/wiki/Shunting_yard_algorithm
31
32while there are tokens to be read:
33read a token
34if the token is:
35- a number:
36put it into the output queue
37- a function:
38push it onto the operator stack
39- an operator o1:
40while (
41there is an operator o2 at the top of the operator stack which is not a left parenthesis,
42and (o2 has greater precedence than o1 or (o1 and o2 have the same precedence and o1 is left-associative))
43):
44pop o2 from the operator stack into the output queue
45push o1 onto the operator stack
46- a left parenthesis (i.e. "("):
47push it onto the operator stack
48- a right parenthesis (i.e. ")"):
49while the operator at the top of the operator stack is not a left parenthesis:
50{assert the operator stack is not empty}
51// If the stack runs out without finding a left parenthesis, then there are mismatched parenthesis.
52pop the operator from the operator stack into the output queue
53{assert there is a left parenthesis at the top of the operator stack}
54pop the left parenthesis from the operator stack and discard it
55if there is a function token at the top of the operator stack, then:
56pop the function from the operator stack into the output queue
57
58// After the while loop, pop the remaining items from the operator stack into the output queue.
59while there are tokens on the operator stack:
60// If the operator token on the top of the stack is a parenthesis, then there are mismatched parenthesis.
61{assert the operator on top of the stack is not a (left) parenthesis}
62pop the operator from the operator stack onto the output queue
63*/

490 vector<string> getRPN(vector<string> tokens) {
491
492 vector<string> input(tokens);
493 vector<string> rpn;
494 stack<string> operators;
495 double val;
496 function<double(double)> func;
497 string op;
498
499 string token;
500
501 size_t i;
502 //Shunting Yard Algorithm
503
504 //while there are tokens to be read:
505
506 for (i = 0; i < input.size(); i++) {
507 //Read a token
508 token = input[i];
509 //If the token is:
510 //a number:
511 if (getNumber(token, val)) {
512 //put it into the output queue
513 rpn.push_back(numberToString(val));
514 }
515 //a function:
516 else if (isFunction(token, func)) {
517 //push it into the operator stack
518 operators.push(token);
519 }
520 //an operator o1:
521 else if (isOperator(token)) {
522 //while (
523 //there is an operator o2 at the top of the operator stack which is not a left parenthesis,
524 //and (o2 has greater precedence than o1 or (o1 and o2 have the same precedence and o1 is left-associative))
525 //)
526 while (!operators.empty() && operators.top() != "("
527 && (precedence(operators.top()) > precedence(token) ||
528 (precedence(operators.top()) == precedence(token) && isLeftAssociative(token)))) {
529 //pop o2 from the operator stack into the output queue
530 op = operators.top();
531 operators.pop();
532
533 rpn.push_back(op);
534 }
535 //push o1 onto the operators stack
536 operators.push(token);
537 }
538 //a left parenthesis i.e. "("
539 else if (token == "(") {
540 // push it onto the operator stack
541 operators.push(token);
542 }
543 //a right parenthesis (i.e. ")")
544 else if (token == ")") {
545 //while the operator at the top of the operator stack is not a left parenthesis:
546 //{assert the operator stack is not empty}
548
549 while (!operators.empty() && operators.top() != "(") {
550 //pop the operator from the operator stack into the output queue
551 op = operators.top();
552 operators.pop();
553 rpn.push_back(op);
554 }
555 //{assert there is a left parenthesis at the top of the operator stack}
556 //pop the left parenthesis from the operator stack and discard it
557
558 if (!operators.empty() && operators.top() == "(") {
559 operators.pop();
560 }
561
562 //if there is a function token at the top of the operator stack, then:
563 //pop the function from the operator stack into the output queue
564 if (!operators.empty() && isFunction(operators.top(), func)) {
565 op = operators.top();
566 operators.pop();
567 rpn.push_back(op);
568 }
569 }
570 //Variable! colocarla en la salida
571 else {
572 //Variable
573 rpn.push_back(token);
574 }
575 }
576
577 /*
578 After the while loop, pop the remaining items from the operator stack into the output queue.
579 while there are tokens on the operator stack:
580 */
581 while (!operators.empty()) {
582 // If the operator token on the top of the stack is a parenthesis, then there are mismatched parenthesis.
583 // Ya se ha verificado el balance de parentesis
584 //{assert the operator on top of the stack is not a (left) parenthesis}
585 //pop the operator from the operator stack onto the output queue
586 op = operators.top();
587 operators.pop();
588 if (op != "(") {
589 rpn.push_back(op);
590 }
591 }
592
593 // rpn almacena la representación RPN de la expresión
594 return rpn;
595 }

672 double evalRPN(vector<string> rpn, vector<Variable> vars) {
673 double result = NAN;
674
675 stack<double> values;
676
677 vector<string> items = rpn;
678 function<double(double)> func;
679 double val;
680
681
682 for (auto & item: rpn) {
683 if (getNumber(item, val)) {
684 values.push(val);
685 }
686 else if (isFunction(item, func)) {
687 //Verificar si existe un valor en la pila
688 if (values.empty()) {
689 return result;
690 }
691 //Obtener el valor del tope de la pila
692 val = values.top();
693 //Eliminar el valor obtenido de la pila
694 values.pop();
695 //Evaluar la funcion y almacenar el resultado en la pila
696 values.push(func(val));
697 }
698 else if (isOperator(item)) {
699 if (isBinaryOperator(item) && values.size() >= 2) {
700 //Obtener b, a de la pila (b esta en el tope)
701 double b = values.top();
702 values.pop();
703 double a = values.top();
704 values.pop();
705 //Calcular y almacenar el resultado en la pila
706 values.push(calculate(a, b, item));
707 }else if (isUnaryOperator(item) && values.size() > 0) {
708 //Obtener un valor del tope de la pila
709 val = values.top();
710 values.pop();
711 //Calcular y almacenar el resultado en la pila
712 values.push(calculateUnary(val, item));
713 }
714 }
715 else if (isVariable(item, vars, val)) {
716 //Almacenar el valor de la variable en la pila
717 values.push(val);
718 }
719 }
720 //Al final de la ejecucion, el tope de la pila contiene el resultado
721 if (values.size() == 1) {
722 result = values.top();
723 values.pop();
724 }
725
726 return result;
727 }
